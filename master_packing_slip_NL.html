const HEADERS = [ 'Closed At', 'BOL #', 'Slip ID', 'Destination', 'Carrier', 'Trailer #', 'Ship Date', 'Pallet Count', 'Total Pieces', 'Pallet IDs', 'SKU Summary' ]; // Tab name mapping — matches what the HTML tool sends as dest const TAB_MAP = { 'DALLAS DC': 'DALLAS DC', 'ORLANDO DC': 'ORLANDO DC', 'ELGIN DC': 'ELGIN DC', }; function doPost(e) { try { const data = JSON.parse(e.postData.contents); appendBolRecord(data); return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // Allow GET for connectivity test (open Web App URL in browser to confirm it's live) function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: 'ok', message: 'BOL Logger is live.' })) .setMimeType(ContentService.MimeType.JSON); } function appendBolRecord(record) { const ss = SpreadsheetApp.openById(SHEET_ID); // Determine which tab to write to const tabName = TAB_MAP[record.dest] || 'OTHER'; let sheet = ss.getSheetByName(tabName); // Create tab if it doesn't exist yet if (!sheet) { sheet = ss.insertSheet(tabName); } // Write headers if sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); // Style header row const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground('#008533'); headerRange.setFontColor('#ffffff'); headerRange.setFontWeight('bold'); sheet.setFrozenRows(1); // Set column widths sheet.setColumnWidth(1, 140); // Closed At sheet.setColumnWidth(2, 120); // BOL # sheet.setColumnWidth(3, 180); // Slip ID sheet.setColumnWidth(4, 100); // Destination sheet.setColumnWidth(5, 100); // Carrier sheet.setColumnWidth(6, 80); // Trailer # sheet.setColumnWidth(7, 90); // Ship Date sheet.setColumnWidth(8, 80); // Pallet Count sheet.setColumnWidth(9, 80); // Total Pieces sheet.setColumnWidth(10, 260); // Pallet IDs sheet.setColumnWidth(11, 320); // SKU Summary } // Build compact pallet ID list const palletIds = (record.pallets || []).map(p => p.id).join(', '); // Build SKU summary: aggregate qty across all pallets per SKU const skuTotals = {}; for (const pallet of (record.pallets || [])) { for (const s of (pallet.skus || [])) { skuTotals[s.sku] = (skuTotals[s.sku] || 0) + s.qty; } } const skuSummary = Object.entries(skuTotals) .map(([sku, qty]) => `${sku}×${qty}`) .join(' / '); // Format closed timestamp const closedAt = record.closedAt ? new Date(record.closedAt).toLocaleString('en-US', { timeZone: 'America/Chicago' }) : ''; // Append row sheet.appendRow([ closedAt, record.bol || '', record.slipId || '', record.dest || '', record.carrier || '', record.trailer || '', record.shipDate || '', record.palletCount || 0, record.totalPieces || 0, palletIds, skuSummary ]); // Alternate row shading for readability const lastRow = sheet.getLastRow(); if (lastRow > 1 && lastRow % 2 === 0) { sheet.getRange(lastRow, 1, 1, HEADERS.length).setBackground('#f0f7f0'); } }
